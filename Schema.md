-> users:
    - bio: str
    - createdAt: datetime
    - email: str
    - avatar: str
    - username: str
    - following: array of strs (ids of followings)
    -> objectives: 
        - lastUpdated: datetime
        - startDate: datetime
        - status: str
        - title: str
    -> projects:
        - createdAt: datetime
        - description: str
        - githubURL: str
        - stack: array of str
        - title: str
        - visibility: str
    -> skills:
        - acquiredAt: datetime
        - level: str
        - name: str
        - updatedAt: datetime

-> userVotes:
    uid: userid_projectid
    vote: string ("upvote" or "downvote")

-> publicProjects:
    - createdAt: datetime
    - stack: array of str
    - title: str
    - uid: str

-> publicFeed:
    - addedAt: datetime
    - content: str
    - pid: str
    - summary: str
    - type: str ("project" or "objective" or "skill" or "text")
    - uid: str
    - votes: int
    - picture: str
    -> comments:
        - content: str
        - createdAt: datetime
        - parentId: str (only if its a reply)
        - uid: str

-> chat:
    -createdAt: timestamp
    -lastUpdate: timestamp
    -lastMessage: map
        - content: str
        - sender_id: str
    -users : array of str
    ->messages
        -content : str
        -created_at : timestamp
        -sender_id : str
